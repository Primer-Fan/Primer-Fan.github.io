NC45编程题

```Java
import java.util.ArrayList;
import java.util.Stack;

/*
 * public class TreeNode {
 *   int val = 0;
 *   TreeNode left = null;
 *   TreeNode right = null;
 * }
 */

public class Solution {
    /**
     * 
     * @param root TreeNode类 the root of binary tree
     * @return int整型二维数组
     */
    public int[][] threeOrders (TreeNode root) {
        // write code here
        int[][] ans = new int[3][];
        ans[0] = pre(root);
        ans[1] = in(root);
        ans[2] = post(root);

        return ans;
    }

    public int[] pre(TreeNode root) {
        ArrayList<Integer> arrayList = new ArrayList<>();
        Stack<TreeNode> stack = new Stack<>();
        while(root != null || stack.isEmpty() == false) {
            while(root != null) {
                arrayList.add(root.val);
                stack.push(root);
                root = root.left;
            }

            if (stack.isEmpty() == false) {
                root = stack.pop();
                root = root.right;
            }
        }
        int[] ans = new int[arrayList.size()];
        for (int i = 0; i < arrayList.size(); ++i) {
            ans[i] = arrayList.get(i);
        }
        return ans;
    }
    
    public int[] in(TreeNode root) {
        ArrayList<Integer> arrayList = new ArrayList<>();
        Stack<TreeNode> stack = new Stack<>();
        while(root != null || stack.isEmpty() == false ) {
            while (root != null) {
                stack.add(root);
                root = root.left;
            }

            if (stack.isEmpty() == false) {
                root = stack.pop();
                arrayList.add(root.val);
                root = root.right;
            }
        }
        int[] ans = new int[arrayList.size()];
        for (int i = 0; i < arrayList.size(); ++i) {
            ans[i] = arrayList.get(i);
        }
        return ans;
    }

    public int[] post(TreeNode root) {
        ArrayList<Integer> arrayList = new ArrayList<>();
        Stack<TreeNode> stack = new Stack<>();
        TreeNode lastVisiNode = root;
        while(root != null || stack.isEmpty() == false) {
            while (root != null) {
                stack.push(root);
                root = root.left;
            }
            root = stack.peek();
            if (root.right == null || root.right == lastVisiNode) {
                arrayList.add(root.val);
                stack.pop();
                lastVisiNode = root;
                root = null;
            }
            else {
                root = root.right;
            }
        }
        int[] ans = new int[arrayList.size()];
        for (int i = 0; i < arrayList.size(); ++i) {
            ans[i] = arrayList.get(i);
        }
        return ans;
    }
}
```
